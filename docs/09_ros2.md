# 09 — ROS2 Integration

## Overview

Two deployment modes depending on your robot's compute:

```
Mode A — HTTP (robot is thin client):
  Camera → ROS2 Node → POST /predict → FastAPI server → masks → publish topic

Mode B — Embedded ONNX (robot runs model locally):
  Camera → ROS2 Node → ONNX Runtime (INT8) → masks → publish topic
```

| Mode | When to use | Latency |
|---|---|---|
| A — HTTP | GPU server on same network, robot is lightweight | ~90ms + network |
| B — Embedded | Robot has Jetson/NUC, no server, edge deployment | ~90ms CPU only |

## 2. Mode B — Embedded ONNX Node (recommended for robot)

```python
# ros2_ws/src/mask2former_ros/mask2former_ros/segmentation_node.py
import rclpy
from rclpy.node import Node
from rclpy.qos import QoSProfile, ReliabilityPolicy
from sensor_msgs.msg import Image
from std_msgs.msg import Header
from cv_bridge import CvBridge

import numpy as np
import cv2
import onnxruntime as ort
from huggingface_hub import hf_hub_download
import os

# Custom message (define in your_msgs package or use vision_msgs)
# from vision_msgs.msg import Detection2DArray, Detection2D, ObjectHypothesisWithPose


class SegmentationNode(Node):

    def __init__(self):
        super().__init__("mask2former_segmentation")

        # Parameters (configurable via config/params.yaml)
        self.declare_parameter("hf_repo_id", "your-hf-username/mask2former-instance-seg")
        self.declare_parameter("model_file", "mask2former_int8.onnx")
        self.declare_parameter("hf_token", "")
        self.declare_parameter("img_size", 512)
        self.declare_parameter("conf_threshold", 0.5)
        self.declare_parameter("input_topic", "/camera/image_raw")
        self.declare_parameter("output_topic", "/perception/instance_masks")
        self.declare_parameter("publish_viz", True)

        hf_repo   = self.get_parameter("hf_repo_id").value
        model_file = self.get_parameter("model_file").value
        hf_token  = self.get_parameter("hf_token").value or None
        self.img_size = self.get_parameter("img_size").value
        self.conf_threshold = self.get_parameter("conf_threshold").value

        # Download and load ONNX model
        self.get_logger().info(f"Loading model from HF Hub: {hf_repo}/{model_file}")
        model_path = hf_hub_download(repo_id=hf_repo, filename=model_file, token=hf_token)
        self.session = ort.InferenceSession(
            model_path,
            providers=["CUDAExecutionProvider", "CPUExecutionProvider"]
        )
        self.get_logger().info("Model loaded successfully.")

        self.bridge = CvBridge()
        self.mean = np.array([0.485, 0.456, 0.406], dtype=np.float32)
        self.std  = np.array([0.229, 0.224, 0.225], dtype=np.float32)

        # Subscriber
        input_topic = self.get_parameter("input_topic").value
        qos = QoSProfile(depth=1, reliability=ReliabilityPolicy.BEST_EFFORT)
        self.sub = self.create_subscription(
            Image,
            input_topic,
            self.image_callback,
            qos,
        )

        # Publishers
        output_topic = self.get_parameter("output_topic").value
        self.mask_pub = self.create_publisher(Image, output_topic, 10)

        if self.get_parameter("publish_viz").value:
            self.viz_pub = self.create_publisher(Image, output_topic + "/visualization", 10)

        self.get_logger().info(f"Subscribed to: {input_topic}")
        self.get_logger().info(f"Publishing to: {output_topic}")

    def image_callback(self, msg: Image):
        try:
            # Convert ROS Image to OpenCV
            cv_image = self.bridge.imgmsg_to_cv2(msg, desired_encoding="rgb8")
            orig_h, orig_w = cv_image.shape[:2]

            # Preprocess
            tensor = self._preprocess(cv_image)

            # Inference
            pred_masks, pred_logits = self.session.run(None, {"image": tensor})

            # Postprocess
            instances = self._postprocess(pred_masks, pred_logits, orig_h, orig_w)

            # Publish mask overlay image
            self._publish_masks(instances, orig_h, orig_w, msg.header)

            # Optional visualization
            if self.get_parameter("publish_viz").value:
                viz = self._draw_visualization(cv_image, instances)
                viz_msg = self.bridge.cv2_to_imgmsg(viz, encoding="rgb8")
                viz_msg.header = msg.header
                self.viz_pub.publish(viz_msg)

        except Exception as e:
            self.get_logger().error(f"Segmentation failed: {e}")

    def _preprocess(self, image: np.ndarray) -> np.ndarray:
        resized = cv2.resize(image, (self.img_size, self.img_size))
        normalized = (resized.astype(np.float32) / 255.0 - self.mean) / self.std
        return normalized.transpose(2, 0, 1)[np.newaxis]

    def _postprocess(self, pred_masks, pred_logits, orig_h: int, orig_w: int):
        instances = []
        scores = pred_logits[0].max(axis=-1)
        class_ids = pred_logits[0].argmax(axis=-1)

        for i, (mask, score, class_id) in enumerate(zip(pred_masks[0], scores, class_ids)):
            if score < self.conf_threshold:
                continue
            mask_resized = cv2.resize(mask.astype(np.float32), (orig_w, orig_h))
            binary_mask = (mask_resized > 0.5).astype(np.uint8)
            instances.append({
                "instance_id": i,
                "category_id": int(class_id),
                "score": float(score),
                "mask": binary_mask,
            })

        return instances

    def _publish_masks(self, instances, orig_h, orig_w, header):
        # Publish combined instance mask (each instance has unique pixel value)
        combined = np.zeros((orig_h, orig_w), dtype=np.uint8)
        for inst in instances:
            combined[inst["mask"].astype(bool)] = inst["instance_id"] + 1

        mask_msg = self.bridge.cv2_to_imgmsg(combined, encoding="mono8")
        mask_msg.header = header
        self.mask_pub.publish(mask_msg)

    def _draw_visualization(self, image: np.ndarray, instances: list) -> np.ndarray:
        viz = image.copy()
        colors = [(255,0,0),(0,255,0),(0,0,255),(255,255,0),(0,255,255),(255,0,255)]
        for inst in instances:
            color = colors[inst["instance_id"] % len(colors)]
            overlay = viz.copy()
            overlay[inst["mask"].astype(bool)] = color
            cv2.addWeighted(overlay, 0.4, viz, 0.6, 0, viz)
        return viz


def main(args=None):
    rclpy.init(args=args)
    node = SegmentationNode()
    try:
        rclpy.spin(node)
    except KeyboardInterrupt:
        pass
    finally:
        node.destroy_node()
        rclpy.shutdown()


if __name__ == "__main__":
    main()
```

## 9. Mode A — HTTP Client (alternative)

```python
# ros2_ws/src/mask2former_ros/mask2former_ros/segmentation_client.py
import requests
import cv2
import numpy as np
from sensor_msgs.msg import Image
from cv_bridge import CvBridge
import rclpy
from rclpy.node import Node


class SegmentationClientNode(Node):
    def __init__(self):
        super().__init__("mask2former_client")
        self.declare_parameter("server_url", "http://localhost:8000")
        self.server_url = self.get_parameter("server_url").value
        self.bridge = CvBridge()
        self.sub = self.create_subscription(Image, "/camera/image_raw", self.callback, 1)

    def callback(self, msg: Image):
        cv_image = self.bridge.imgmsg_to_cv2(msg, "bgr8")
        _, buffer = cv2.imencode(".jpg", cv_image)

        response = requests.post(
            f"{self.server_url}/predict",
            files={"file": ("frame.jpg", buffer.tobytes(), "image/jpeg")},
            timeout=5.0,
        )

        if response.ok:
            data = response.json()
            self.get_logger().info(
                f"Detected {len(data['instances'])} instances in {data['inference_ms']:.1f}ms"
            )
```

## Summary

| Component | Details |
|---|---|
| Node language | Python (rclpy) |
| Input | `/camera/image_raw` (sensor_msgs/Image) |

| Output | `/perception/instance_masks` (uint8 mask) |
| Visualization | `/perception/instance_masks/visualization` (RGB overlay) |
| ROS2 version | Humble / Iron compatible |
